    06. switch문

     {
    const num = 100;

    switch(num){
        case 100 :  
            document.write("실행(num==100)");
        break; 
        case 200 :  
            document.write("실행(num==200)");
        break; 
        case 300 :  
            document.write("실행(num==300)");
        break; 
        default :  
            document.write("실행(num==값이없음)");
        break;    
    }
}

    결과보기

    실행(num==100)

    12. continue문

     {
    let num = 0;

    while(num<10){ 
        num++;                        
        if(num == 5){
            continue;
        }      
        document.write(num);          
    }
    for(let i=1; i<=10; i++){
        if(i == 5){
            continue;
        }
        document.write(i);
    }   
}
