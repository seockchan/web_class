반복문

반복문을 이용하면 코드를 원하는 횟수만큼 반복하여 실행할 수 있습니다.

while문

while문은 조건식을 만족할 때까지 코드를 여러 회 반복하여 실행할 수 있습니다.


다음은 while 문의 기본형입니다. while 문은 조건식을 만족할 때까지 중괄호 안에 있는 코드를 반복하여 실행합니다. while 문의 실행 순서는 조건식을 검사하고, 만족하면 중괄호 안에 있는 코드와 증감식을 실행합니다. 그리고 다시 조건식을 검사합니다.

    let 변수 = 초깃값;
    while(조건식){
        자바스크립트 코드;
        증감식;
    }

예제

    var i = 1;
    while(i<=30){        //i가 30까지 반복
        if(i%2 == 0&& i%6 == 0){        // i가 2의 배수거나 6의 배수 이면 출력
            document.write(i,"<br>")  
        }
        i++
    }   //6,12,18,24,30출력

for문

for문은 조건식을 만족할 때까지 특정 코드를 반복하여 실행합니다. 사용 방법은 while문과 같지만 while문보다 사용하기 편해 사용 빈도가 높은 편입니다.

   for(초깃값; 조건식; 증감식){
    자바스크립트 코드;
    }

break문

반복문인 while문과 for문에서 break문을 실행하면 조건식과 상관없이 강제로 반복문을 종료합니다. 즉, break문은 반복문을 강제로 종료할 때 사용합니다.


다음은 for문과 while문에서 break문이 사용된 기본형입니다. break문이 코드보다 앞에 있으므로 코드는 실행되지 않고 for문과 while문이 바로 종료됩니다.

   for(초깃값; 조건식; 증감식){
    break; //   반복문을 강제로 종료합니다.
    자바스크립트 코드;
    }

예제

    var i = 1;
    for(var i=1; i<=10; i++){ //i가 10까지 반복
        if(i==6)break;          //6이 되면 멈춤
        document.write(i,"<br>")
    }    document.write("===The End===")        //1,2,3,4,5,TheEnd출력

continue문

continue문은 반복문에서만 사용할 수 있습니다. while문에 사용할 경우 contine문 다음에 오는 코드는 무시하고 바로 조건식으로 이동해 조건 검사를 합니다.


for문에서 continue문을 실행할 경우에는 continue문 다음에 오는 코드는 무시하고 바로 증감식으로 이동하여 증감 연산을 실행합니다.

   for(초깃값; 조건식; 증감식){
    continue;
    자바스크립트코드;
    }  //for문의 continue

    let 변수 = 초깃값;
    while(조건식){
        증감식;
        continue;
        자바스크립트 코드;  
    }    //while문의 continue

예제

    for(var i =1; i<= 10; i++){    //i가 10이 될 때까지 반복
    if(i%2 == 0)continue;   //  2의 배수이면 continue문 실행 
    document.write(i,"<br>"); // 홀수만 출력
    }
