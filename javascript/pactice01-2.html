1. 구구단 출력하기

1. for문을 사용해서 2단부터 9단까지 출력하시오!

{
    for(let i=2; i<=9; i++){
        document.write("<br>")
        for(let j=1; j<=9; j++){
            let sum = i*j;
            document.write(i + "*" + j + "=" + sum,"<br>");
        }
    }
}

1. while문을 사용해서 2단부터 9단까지 출력하시오!

{
    {
        let num1 = 2;
        while(num1<=9){
            document.write("<br>");
            let num2 = 1;
            while(num2<=9){
                let sum = num1*num2;
                document.write(num1 + "*" + num2 + "=" + sum,"<br>");
                num2++;
            }
            num1++;
        }

    }
}
